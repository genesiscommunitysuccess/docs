# Text Field

Use cases:
* User registration and login
* Data entry
* Search boxes
* Comment sections
* Search filters

## Example

<TextFieldDemo />

<Tabs defaultValue="genesis" values={[{ label: 'Genesis', value: 'genesis', }, { label: 'React', value: 'react', }, { label: 'Angular', value: 'angular', }]}>

  <TabItem value="genesis">

    Declaration
    ```html
    <rapid-text-field>Sample Text Field</rapid-text-field>
    ```

    Usage
    ```typescript
    @customElement({
      name: 'my-element',
      template: html`
        <rapid-text-field :value=${sync((x) => x.fieldValue)}>
          Sample Text Field
        </rapid-text-field>
      `,
    })
    export class MyElement extends GenesisElement {
      @observable fieldValue = '';
      fieldValueChanged() {
        console.log(this.fieldValue);
      }
    }
    ```

  </TabItem>
  <TabItem value="react">

    Declaration
    ```html
    <rapid-text-field></rapid-text-field>
    ```
    Usage
    ```typescript
    export function MyComponent() {
      const handleTextFieldChanged = (event) => {
        console.log(event.target.value)
      };
      return (
        <rapid-text-field onChange={handleTextFieldChanged}></rapid-text-field>
      );
    }
    ```
  </TabItem>
  <TabItem value="angular">

    Declaration
    ```html
    <rapid-text-field></rapid-text-field>
    ```

    Usage
    ```typescript
    import { Component, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
    import { FormsModule } from '@angular/forms';
    @Component({
      selector: 'my-root',
      template: `
        <rapid-text-field
          (change)="onTextFieldChange($event)"
        ></rapid-text-field>
      `,
      styleUrls: ['./my.component.css'],
      standalone: true,
      schemas: [CUSTOM_ELEMENTS_SCHEMA],
      imports: [FormsModule],
    })
    export class AppComponent {
      onTextFieldChange(event: Event) {
        const target = event.target as HTMLInputElement;
        console.log(target.value);
      }
    }
    ```
  </TabItem>
</Tabs>

### Properties

This component doesn't have any properties which are not also controlled via attributes.

### Slots

| Slot | Use |
|---|---|
| start | Start slot. |
| control | The input element. |
| end | End slot. |

#### Example
Numeric amount using `$` symbol in start slot
<TextFieldWithSlot />

Implementation
```html
<rapid-text-field type="text" inputmode="numeric" pattern="\d*">
    <div slot="start">
        $
    </div>
</rapid-text-field>
```
